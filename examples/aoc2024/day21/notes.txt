Pitäisi painella <^
Liikutaan A:sta alas ja ihan vasemmalle, sitten ylös: v<<A>^A

numpad moves:
	"^^" => ""
	"^A" => ">"
	"^<" => "v<"
	"^v" => "v"
	"^>" => "v>" tai ">v"

	"A^" => "<"
	"AA" => ""
	"A<" => "v<<"
	"Av" => "v<"
	"A>" => "v"

	"<^" => ">^"
	"<A" => ">>^"
	"<<" => ""
	"<v" => ">"
	"<>" => ">>"

	"v^" => "^"
	"vA" => ">^" tai "^>"
	"v<" => "<"
	"vv" => ""
	"v>" => ">"

	">^" => "<^" tai "^<"
	">A" => "^"
	"><" => "<<"
	">v" => "<"
	">>" => ""
