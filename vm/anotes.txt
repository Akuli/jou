  -m 1G \
  -bios /usr/share/qemu-efi-aarch64/QEMU_EFI.fd\
  -drive if=virtio,format=raw,file=alpine-standard-3.23.2-aarch64.iso,media=cdrom \

  -nic user,hostfwd=tcp:127.0.0.1:2222-:2222

Must select serial0 in qemu window to see anything
Default is the qemu monitor which is not really useful here.

setup-alpine -c answerfile
sed -i s:'DISKOPTS=none':'DISKOPTS="-m sys /dev/vdb"': answerfile
setup-alpine -e -f answerfile       # kysyy kirjoitetaanko levylle
setup-alpine -e -q -f answerfile    # taisi ignorata answerfilen

echo 'iface eth0 inet dhcp' > /etc/network/interfaces
ifup eth0
echo https://dl-cdn.alpinelinux.org/alpine/v3.23/main > /etc/apk/repositories
apk update
apk add util-linux
